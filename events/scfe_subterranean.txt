namespace = scfe_subterranean

country_event = {
    id = scfe_subterranean.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        NOT = { has_global_flag = scfe_subterranean_started }
        is_ai = no
        years_passed > 40
        count_owned_planet = {
            limit = {
                is_colony = yes
                is_capital = no
                is_artificial = no
                NOR = {
                    exists = archaeological_site
                    has_planet_flag = colony_event
                }
            }
            count > 3
        }
    }

    immediate = {
        set_global_flag = scfe_subterranean_started
        random_owned_planet = {
            limit = {
                is_colony = yes
                is_capital = no
                is_artificial = no
                NOR = {
                    exists = archaeological_site
                    has_planet_flag = colony_event
                }
            }
            save_global_event_target_as = scfe_subterranean_planet
            set_planet_flag = colony_event
        }
        if = {
            limit = {
                any_owned_planet = { has_planet_flag = seismic_disturbance_friendly }
            }
            set_country_flag = seismic_friendly
            country_event = { id = scfe_subterranean.3 days = 200 random = 200 }
        }
        else = {
            country_event = { id = scfe_subterranean.2 days = 200 random = 200 }
        }
    }
}
