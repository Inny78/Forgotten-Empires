            set_faction_hostility = { set_hostile = no }

                    trait = trait_machine_unit
                    trait = random_traits

                immortal = yes

                class = official

                    trait = leader_trait_mining_focus

                event_leader = yes
                hide_age = yes
                skip_background_generation = yes

        }
