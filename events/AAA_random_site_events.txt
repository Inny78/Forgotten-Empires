# Unstable Construction
fleet_event = {
	id = AAA_random_arcsite.2
	title = "AAA_random_arcsite.2.name"
	desc = "AAA_random_arcsite.2.desc"
	picture = GFX_evt_derelict_interior
	show_sound = event_energy_field

	archaeology = yes

	is_triggered_only = yes

	immediate = {
		fromfrom = { set_site_progress_locked = yes }
		leader = {
			save_event_target_as = maimed_scientist
			add_trait = { trait = leader_trait_maimed }
		}
	}

	after = {
		fromfrom = { set_site_progress_locked = no }
	}

	option = {
		name = site_success
		fromfrom = { add_stage_clues = -2 }
	}
}
