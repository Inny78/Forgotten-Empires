@precursor_easy = 6

alshigans_1_cat = {
    desc = alshigans_1_category_desc
    picture = GFX_evt_ship_in_orbit
    level = @precursor_easy

    spawn_chance = {
        modifier = {
            add = 10
            solar_system = { has_star_flag = precursor_alshigans }
            is_asteroid = yes
            NOT = { has_planet_flag = precursor_world }
            from.owner = {
                NOT = { has_country_flag = alshigans_1 }
                has_country_flag = alshigans_intro
                has_event_chain = scfe_alshigans_chain
            }
        }
    }

    on_spawn = {
        set_planet_flag = precursor_world
        from.owner = {
            set_country_flag = alshigans_1
            save_global_event_target_as = scfe_alshigans_owner
        }
        set_planet_flag = scfe_alshigans_start
        save_global_event_target_as = scfe_alshigans_asteroid
    }
    on_success = scfe_precursor_init.2
}
