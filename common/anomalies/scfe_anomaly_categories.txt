	desc = scfe_belltower_desc
	picture = GFX_evt_scfe_dunes
	level = 8
	max_once_global = yes
	spawn_chance = {
		modifier = {
			add = 1
			solar_system = {

				}
			}

		modifier = {
			factor = 0

				OR = {

				}
			}
		}
	}
	on_success = scfe_belltower.1
}
