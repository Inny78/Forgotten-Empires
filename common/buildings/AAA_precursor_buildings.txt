# Archaeotech
# For plexian clone vats: If machine empire, add scientist jobs and/or archeoresearcher jobs, unless you have the catalytic processing civic, in which case it will provide mechanical pop assembly.
building_plexian_clone_vats = {
	icon = building_ancient_relics
	base_buildtime = @b2_time
	base_cap_amount = 1
	category = pop_assembly

    building_sets = {
        government
        urban
    }

    potential = {
		exists = owner
		owner = { has_technology = tech_secrets_plexian }
		NOT = { has_modifier = resort_colony }
		OR = {
			owner = { is_ai = no }
			NOT = { exists = assembling_species }
			assembling_species = { is_organic_species = yes }
		}
		OR = {
			owner = { is_ai = no }
			NAND = {
				free_district_slots = 0
				free_building_slots <= 1
				free_housing <= 0
				free_jobs <= 0
			}
		}
	}

	allow = {
		has_major_upgraded_capital = yes
	}

	destroy_trigger = {
		exists = owner
		OR = {
			AND = {
				owner = { is_ai = yes }
				free_district_slots = 0
				free_building_slots = 0
				free_housing <= 0
				free_jobs <= 0
			}
			has_modifier = resort_colony
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			minor_artifacts = 30
		}
		# Base upkeep
		upkeep = {
			energy = 5
		}
		# Org/Lith upkeep
		upkeep = {
			trigger = {
				exists = owner
				owner = { is_lithoid_empire = no }
				OR = {
					owner = { is_machine_empire = no }
					owner = { has_civic = civic_machine_catalytic_processing }
				}
			}
			food = 30
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { is_lithoid_empire = yes }
			}
			minerals = 30
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = no }
		}
        bonus_pop_growth = 2

	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = no }
			owner = { has_ascension_perk = ap_archaeoengineers }
		}
		modifier = {
            bonus_pop_growth = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
			NOT = { owner = { has_civic = civic_machine_catalytic_processing } }
		}
		modifier = {
			job_coordinator_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
			NOT = { owner = { has_civic = civic_machine_catalytic_processing } }
			owner = { has_ascension_perk = ap_archaeoengineers }
		}
		modifier = {
			job_coordinator_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
			owner = { has_civic = civic_machine_catalytic_processing }
		}
		modifier = {
			planet_pop_assembly_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
			owner = { has_civic = civic_machine_catalytic_processing }
			owner = { has_ascension_perk = ap_archaeoengineers }
		}
		modifier = {
			planet_pop_assembly_add = 1
		}
	}

	on_built = {
		set_planet_flag = can_organic_assemble_flag
	}

	on_destroy = {
		remove_planet_flag = can_organic_assemble_flag
	}

	prerequisites = {
		"tech_secrets_plexian"
	}
}
